	\section*{Что можно делать в $\mathbb{R}$}
	
	$ \sqrt{2} := \sup\{r \in \mathbb{Q} \; |\; ^{n > 0}_{n^{2} < 2} \} $\\
	\\
	$ \sqrt[n]{a} := \sup\{r \in \mathbb{Q} \; |\; r^{n} < a \} $\\
	\\
	$a^{b} := \sup\{a^{r} \; | \; r \in \mathbb{Q}; \; r < b \} $\\
	$a \ge 0, \; b \in \mathbb{R}$
